1、实现： 

sum(1, 2, 3).sumOf(); //6 
sum(2, 3)(2).sumOf(); //7 
sum(1)(2)(3)(4).sumOf(); //10 
sum(2)(4, 1)(2).sumOf(); //9 

```js
function sum(...args) {
    const arg = [...args]

    count = function(...args) {
        arg.push(...args)
        return count
    }
    count.sumOf = function() {
        return arg.reduce((a, b) => a + b)
    }
    return count
}
```
