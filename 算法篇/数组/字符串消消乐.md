// 示例：abcccbxezzzrf7788fn
// 输出：axern
function delStr(str) {
  const stack = []
  for(let i = 0; i < str.length; i++) {
    const char = str[i]
    if(str[i+1] === char || char === str[i-1]) continue // 处理连续的子串
    if(stack.length && stack[stack.length - 1] === char) {
      stack.pop()
    } else {
      stack.push(char)
    }
  }
  return stack.join('')
}
console.log(delStr('abcccbxezzzrf7788fn'))
